### 队列
队列一个最基本的原则就是先进先出
可以用数组和链表实现
用链表的话就是一个头指针和一个尾指针分别指向头节点和尾节点
用数组的话就是循环使用数组，如果队列满了也可以选择扩充队列

因为是循环队列，我们的判断条件会有点不一样
队列头为front，队列尾为rear, 队列长度为10

若是判断队列是否满
```c++
//为什么是rear + 1, 因为这是循环队列
//假设front = 0, rear = 9;
//rear不可能超过数组长度，所以为了将其返回到头，则要取余数
//(rear + 1) % length 正好等于front
if((rear + 1) % length == front)
{
    cout << "队列已满" << endl;
}
```

若是判断队列是否空
```c++
if(rear == front)
{
    cout << "队列为空" << endl;
}
```

> 这里使用了模板编程，所以所有模板及其模板方法声明定义都在头文件内
```c++
"queue.h"
#ifndef _QUEUE_H
#define _QUEUE_H

#include <iostream>

using namespace std;

template<typename T>
class Queue
{
private:
    T *array; //数组
    int length; //数组长度
    int front; //队列�?
    int rear; //队列�?
public:
    void InQueue(T element);
    T DeQueue();
    void Show();
    Queue();
    Queue(int length);
    ~Queue();
};

template<typename T>
Queue<T>::Queue()
{
    array = new T[1];//默认分配一点就�?
    array[0] = 0;
    length = 0;
    front = 0;
    rear = 0;
}

template<typename T>
Queue<T>::Queue(int length)
{
    cout << "Queue Initializing" << endl;
    array = new T[length];//默认分配一点就行
    for(int i = 0 ; i < length; i++)//初始化
    {
        array[i] = 0;
    }
    length = length;
    front = 0;
    rear = 0;
    cout << "Queue finished Initialize" << endl;
}

template<typename T>
Queue<T>::~Queue()
{
    delete [] array;//必须和new一样
}

template<typename T>
void Queue<T>::InQueue(T element)
{
    if((rear+1)%length == front)
    {
        cout << "队列已满" << endl;
    }
    array[rear] = element;
    rear = (rear + 1) % length;
}

template<typename T>
T Queue<T>::DeQueue()
{
    if(rear == front)
    {
        cout << "Queue is empty" << endl;
    }
    int deQueueElement = array[front];
    front = (front + 1) % length;
    return deQueueElement;
}

template<typename T>
void Queue<T>::Show()
{
    for(int i = front; i != rear; i = (i+1) % length)
    {
        cout << array[i] << " ";
    }
    cout << endl;
}

#endif

"main.cpp"
#include <iostream>
#include "queue.h"

using namespace std;

int main(int argc, char const *argv[])
{
    Queue<int> que(10);
    que.InQueue(1);
    que.InQueue(2);
    cout << que.DeQueue() << endl;
    que.Show();
    return 0;
}
```
